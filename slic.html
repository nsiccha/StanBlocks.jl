HSGP building block (hsgp below)

SlicModel
  • Specification
  • Data
  • Stan
begin
    xi = (2 * (x - min(x))) / (max(x) - min(x)) - 1.0
    L = 1.5
    n_functions = 20
    X = sin((pi / (2L)) * (xi + L) * (1:n_functions)') / sqrt(L)
    x_scale ~ uniform(0, 2)
    y_scale ~ lognormal(0, 1)
    scale = y_scale * sqrt(sqrt(2pi) * x_scale) * exp(-0.25 * ((x_scale * pi) / (2L)) ^ 2 * (1:n_functions) ^ 2)
    unit_weight ~ std_normal(; n = n_functions)
    return X * (scale .* unit_weight)
end

Qualitatively reproduces Charles C. Margossian’s planetary motion case study.

Only one model definition is needed.

SlicModel
  • Specification
  • Data
  • Stan
begin
    "Unsure what function definition syntax should look like: `ode(...) = ...`"
    t0 = 0.0
    m = 1.0
    sigma_x = sigma
    sigma_y = sigma
    rel_tol = 1.0e-6
    abs_tol = 1.0e-6
    max_steps = 1000
    "The below actually is a very informative prior."
    k ~ normal(1, 0.001; lower = 0.0)
    p01 ~ normal(0, 1)
    p02 ~ lognormal(0, 1)
    "Should Stan's `array` definitions abuse Julia's Set definition syntax?"
    p0 = {p01, p02}
    q0 ~ normal(0, 1; n = 2)
    star ~ normal(0, 0.5)
    y0 = append_array(q0, p0)
    theta = append_array({k}, star)
    y = integrate_ode_rk45(ode, y0, t0, time, theta, {m}, x_i, rel_tol, abs_tol, max_steps)
    q_obs ~ normal(y, sigma)
end

Qualitatively reproduces Bob Carpenter’s multiple species-site occupancy case study.

SlicModel
  • Specification
  • Data
  • Stan
begin
    logit_psi = uv[:, 1] + alpha
    logit_theta = uv[:, 2] + beta
    alpha ~ cauchy(0, 2.5)
    beta ~ cauchy(0, 2.5)
    sigma_uv ~ cauchy(0, 2.5)
    rho_uv_beta ~ beta(2, 2)
    rho_uv = 2rho_uv_beta - 1
    uv ~ multi_normal(rep_vector(0, 2), cov_matrix_2d(sigma_uv, rho_uv))
    Omega ~ beta(2, 2)
    for i = 1:n
        1 ~ bernoulli(Omega)
        for j = 1:J
            if x[i, j] > 0
                target += lp_observed(x[i, j], K, logit_psi[i], logit_theta[i])
            else
                target += lp_unobserved(K, logit_psi[i], logit_theta[i])
            end
        end
    end
    for i = n + 1:S
        target += lp_never_observed(J, K, logit_psi[i], logit_theta[i], Omega)
    end
    E_N = S * Omega
    E_N_2 = n
    for i = n + 2:S
        lp_unavailable = bernoulli_lpmf(0, Omega)
        lp_available = bernoulli_lpmf(1, Omega) + J * lp_unobserved(K, logit_psi[i], logit_theta[i])
        Pr_available = exp(lp_available - log_sum_exp(lp_unavailable, lp_available))
        E_N_2 = E_N_2 + bernoulli_rng(Pr_available)
    end
    sim_uv = multi_normal_rng(rep_vector(0, 2), cov_matrix_2d(sigma_uv, rho_uv))
    logit_psi_sim = alpha + sim_uv[1]
    logit_theta_sim = beta + sim_uv[2]
end
